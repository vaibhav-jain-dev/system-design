If none found, wrap to the smallest token — that's the ring property."}} {{code "python" `import hashlib import bisect from collections import defaultdict class ConsistentHashRing: def __init__(self, virtual_nodes: int = 150): self.virtual_nodes = virtual_nodes self.ring: dict[int, str] = {} # token -> node_id self.sorted_keys: list[int] = [] # sorted token list def _hash(self, key: str) -> int: return int(hashlib.md5(key.encode()).hexdigest(), 16) % (2**32) def add_node(self, node_id: str) -> None: for i in range(self.virtual_nodes): token = self._hash(f"{node_id}#{i}") self.ring[token] = node_id bisect.insort(self.sorted_keys, token) def remove_node(self, node_id: str) -> None: for i in range(self.virtual_nodes): token = self._hash(f"{node_id}#{i}") del self.ring[token] idx = bisect.bisect_left(self.sorted_keys, token) self.sorted_keys.pop(idx) def get_node(self, key: str) -> str | None: if not self.ring: return None token = self._hash(key) idx = bisect.bisect(self.sorted_keys, token) # Wrap around: if past the last token, use the first if idx == len(self.sorted_keys): idx = 0 return self.ring[self.sorted_keys[idx]] def get_nodes(self, key: str, count: int) -> list[str]: """Return count distinct physical nodes for replication.""" if not self.ring: return [] token = self._hash(key) idx = bisect.bisect(self.sorted_keys, token) nodes_seen: set[str] = set() result: list[str] = [] for _ in range(len(self.sorted_keys)): node = self.ring[self.sorted_keys[idx % len(self.sorted_keys)]] if node not in nodes_seen: nodes_seen.add(node) result.append(node) if len(result) == count: break idx += 1 return result # Usage ring = ConsistentHashRing(virtual_nodes=150) ring.add_node("node-1") ring.add_node("node-2") ring.add_node("node-3") print(ring.get_node("user:12345")) # e.g. "node-2" print(ring.get_nodes("user:12345", 3)) # ["node-2", "node-1", "node-3"] (replication) `}} {{hint "bisect_left vs bisect" "Use bisect.bisect (= bisect_right) so that if the key hashes exactly to a token, it lands on the NEXT node clockwise. bisect_left would put it on the token itself — either works as long as you're consistent."}} {{hint "sorted_keys maintenance cost" "bisect.insort is O(N) for list insertion because Python's list requires shifting elements.
