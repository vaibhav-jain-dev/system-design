O(log(N * vnodes)) via binary search.""" if not self.ring: return None h = self._hash(key) idx = bisect.bisect_right(self.sorted_keys, h) if idx == len(self.sorted_keys): idx = 0 # wrap around return self.ring[self.sorted_keys[idx]] def get_replicas(self, key: str, count: int = 3) -> list: """Walk clockwise collecting distinct physical nodes.""" if not self.ring: return [] h = self._hash(key) idx = bisect.bisect_right(self.sorted_keys, h) replicas, seen = [], set() for _ in range(len(self.sorted_keys)): if idx == len(self.sorted_keys): idx = 0 node = self.ring[self.sorted_keys[idx]] if node not in seen: replicas.append(node) seen.add(node) if len(replicas) == count: break idx += 1 return replicas # --- Usage --- ring = ConsistentHashRing(["cache-1", "cache-2", "cache-3"]) for key in ["user:1001", "user:1002", "session:abc"]: print("{} -> {} replicas: {}".format( key, ring.get_node(key), ring.get_replicas(key, 2))) ring.add_node("cache-4") # only ~1/4 of keys move ring.remove_node("cache-2") # only its keys move clockwise `}} {{hint "get_replicas skips same physical node" "The get_replicas method collects distinct physical nodes, not distinct virtual nodes.
