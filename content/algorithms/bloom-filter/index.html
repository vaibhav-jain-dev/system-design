Bloom filter wins by 30-50x on memory, which is why it is used wherever exact membership is not required."}} {{phase 4 "Full Python Implementation" "8 min"}} {{code "python" `import math import hashlib class BloomFilter: """Bloom filter with configurable false positive rate.""" def __init__(self, expected_elements: int, fp_rate: float = 0.01): self.n = expected_elements self.p = fp_rate # Optimal bit array size: m = -(n * ln(p)) / (ln(2))^2 self.m = int(-expected_elements * math.log(fp_rate) / (math.log(2) ** 2)) # Optimal hash count: k = (m/n) * ln(2) self.k = max(1, int((self.m / expected_elements) * math.log(2))) self.bits = bytearray((self.m + 7) // 8) self.count = 0 def _get_bit(self, index: int) -> bool: return bool(self.bits[index // 8] & (1 << (index % 8))) def _set_bit(self, index: int): self.bits[index // 8] |= (1 << (index % 8)) def _hashes(self, item: str) -> list: """Double hashing: h(i) = (h1 + i*h2) % m for k positions.""" h1 = int.from_bytes(hashlib.md5(item.encode()).digest()[:8], "big") h2 = int.from_bytes(hashlib.md5((item + ":salt").encode()).digest()[:8], "big") return [(h1 + i * h2) % self.m for i in range(self.k)] def add(self, item: str): """Insert element. O(k) time.""" for pos in self._hashes(item): self._set_bit(pos) self.count += 1 def might_contain(self, item: str) -> bool: """False = definitely absent. True = probably present.""" return all(self._get_bit(pos) for pos in self._hashes(item)) def estimated_fp_rate(self) -> float: return (1 - math.exp(-self.k * self.count / self.m)) ** self.k # --- Usage --- bf = BloomFilter(expected_elements=1_000_000, fp_rate=0.01) print("Bits: {:,} ({:.1f} MB), Hash fns: {}".format(bf.m, bf.m/8/1e6, bf.k)) for i in range(100_000): bf.add("item:{}".format(i)) assert bf.might_contain("item:0") == True assert bf.might_contain("item:99999") == True false_pos = sum(1 for i in range(100_000) if bf.might_contain("missing:{}".format(i))) print("FP rate: {:.3f}% (expected: {:.3f}%)".format( 100 * false_pos / 100_000, 100 * bf.estimated_fp_rate())) `}} {{phase 5 "Use Cases in System Design" "5 min"}} {{say "Bloom filters appear in many system design problems as an optimization layer.
