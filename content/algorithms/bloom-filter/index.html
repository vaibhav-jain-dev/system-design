O(k) time.""" for pos in self._hashes(item): self._set_bit(pos) self.count += 1 def might_contain(self, item: str) -> bool: """False = definitely absent. True = probably present.""" return all(self._get_bit(pos) for pos in self._hashes(item)) def estimated_fp_rate(self) -> float: return (1 - math.exp(-self.k * self.count / self.m)) ** self.k # --- Usage --- bf = BloomFilter(expected_elements=1_000_000, fp_rate=0.01) print("Bits: {:,} ({:.1f} MB), Hash fns: {}".format(bf.m, bf.m/8/1e6, bf.k)) for i in range(100_000): bf.add("item:{}".format(i)) assert bf.might_contain("item:0") == True assert bf.might_contain("item:99999") == True false_pos = sum(1 for i in range(100_000) if bf.might_contain("missing:{}".format(i))) print("FP rate: {:.3f}% (expected: {:.3f}%)".format( 100 * false_pos / 100_000, 100 * bf.estimated_fp_rate())) `}} {{hint "double hashing" "The _hashes method runs only 2 MD5 computations regardless of k. It derives all k positions from h1 and h2 using the formula (h1 + i*h2) % m.
