You do NOT need a background thread adding tokens every millisecond. On each request, compute how many tokens WOULD have been added since the last check. This makes the algorithm stateless between requests and trivial to implement in Redis."}} {{phase 3 "Python Implementation (Thread-Safe)" "8 min"}} {{say "Here is a complete thread-safe implementation suitable for a single-server deployment. For distributed systems, see the Redis version in Phase 5."}} {{code "python" `import time import threading class TokenBucket: """Thread-safe token bucket rate limiter.""" def __init__(self, capacity: float, refill_rate: float): self.capacity = capacity self.refill_rate = refill_rate self.tokens = capacity self.last_refill = time.monotonic() self.lock = threading.Lock() def _refill(self): now = time.monotonic() elapsed = now - self.last_refill self.tokens = min(self.capacity, self.tokens + elapsed * self.refill_rate) self.last_refill = now def allow(self, tokens_needed: int = 1) -> dict: with self.lock: self._refill() if self.tokens >= tokens_needed: self.tokens -= tokens_needed return {"allowed": True, "remaining": int(self.tokens), "retry_after": 0} deficit = tokens_needed - self.tokens return {"allowed": False, "remaining": 0, "retry_after": round(deficit / self.refill_rate, 2)} class RateLimiter: """Manages one token bucket per user/key.""" def __init__(self, capacity: float, refill_rate: float): self.capacity = capacity self.refill_rate = refill_rate self.buckets = {} self.lock = threading.Lock() def is_allowed(self, key: str) -> dict: with self.lock: if key not in self.buckets: self.buckets[key] = TokenBucket(self.capacity, self.refill_rate) return self.buckets[key].allow() # --- Usage --- limiter = RateLimiter(capacity=10, refill_rate=2.0) for i in range(15): result = limiter.is_allowed("user:123") status = "OK" if result["allowed"] else "REJECTED" print("Request {}: {} (remaining: {})".format(i + 1, status, result["remaining"])) # First 10 ALLOW, then 5 REJECT. Wait 5s -> 10 tokens refill. `}} {{phase 4 "Algorithm Comparison" "5 min"}} {{say "There are four main rate limiting algorithms. A Lua script ensures atomic check-and-update with no race conditions."}} {{code "python" `import time import redis class DistributedTokenBucket: """Distributed token bucket using Redis + Lua for atomicity.""" LUA_SCRIPT = """ local key = KEYS[1] local capacity = tonumber(ARGV[1]) local refill_rate = tonumber(ARGV[2]) local now = tonumber(ARGV[3]) local requested = tonumber(ARGV[4]) local bucket = redis.call('HMGET', key, 'tokens', 'last_refill') local tokens = tonumber(bucket[1]) or capacity local last_refill = tonumber(bucket[2]) or now -- Lazy refill local elapsed = math.max(0, now - last_refill) tokens = math.min(capacity, tokens + elapsed * refill_rate) local allowed = 0 local retry_after = 0 if tokens >= requested then tokens = tokens - requested allowed = 1 else retry_after = math.ceil((requested - tokens) / refill_rate) end redis.call('HMSET', key, 'tokens', tokens, 'last_refill', now) redis.call('EXPIRE', key, math.ceil(capacity / refill_rate) * 2) return {allowed, math.floor(tokens), retry_after} """ def __init__(self, redis_client: redis.Redis): self.redis = redis_client self.script = self.redis.register_script(self.LUA_SCRIPT) def is_allowed(self, key: str, capacity: int, refill_rate: float, tokens_needed: int = 1) -> dict: now = time.time() result = self.script( keys=[key], args=[capacity, refill_rate, now, tokens_needed]) return {"allowed": bool(result[0]), "remaining": int(result[1]), "retry_after": int(result[2]), "limit": capacity} # --- Usage across multiple API servers --- r = redis.Redis(host="localhost", port=6379) limiter = DistributedTokenBucket(r) result = limiter.is_allowed("rl:user:42", capacity=100, refill_rate=10) if result["allowed"]: print("Allowed, {} remaining".format(result["remaining"])) else: print("Rate limited, retry in {}s".format(result["retry_after"])) `}} {{avoid "Never implement distributed rate limiting with separate GET and SET commands.
