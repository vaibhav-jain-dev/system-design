For distributed systems, see the Redis version in Phase 5."}} {{code "python" `import time import threading class TokenBucket: """Thread-safe token bucket rate limiter.""" def __init__(self, capacity: float, refill_rate: float): self.capacity = capacity self.refill_rate = refill_rate self.tokens = capacity self.last_refill = time.monotonic() self.lock = threading.Lock() def _refill(self): now = time.monotonic() elapsed = now - self.last_refill self.tokens = min(self.capacity, self.tokens + elapsed * self.refill_rate) self.last_refill = now def allow(self, tokens_needed: int = 1) -> dict: with self.lock: self._refill() if self.tokens >= tokens_needed: self.tokens -= tokens_needed return {"allowed": True, "remaining": int(self.tokens), "retry_after": 0} deficit = tokens_needed - self.tokens return {"allowed": False, "remaining": 0, "retry_after": round(deficit / self.refill_rate, 2)} class RateLimiter: """Manages one token bucket per user/key.""" def __init__(self, capacity: float, refill_rate: float): self.capacity = capacity self.refill_rate = refill_rate self.buckets = {} self.lock = threading.Lock() def is_allowed(self, key: str) -> dict: with self.lock: if key not in self.buckets: self.buckets[key] = TokenBucket(self.capacity, self.refill_rate) return self.buckets[key].allow() # --- Usage --- limiter = RateLimiter(capacity=10, refill_rate=2.0) for i in range(15): result = limiter.is_allowed("user:123") status = "OK" if result["allowed"] else "REJECTED" print("Request {}: {} (remaining: {})".format(i + 1, status, result["remaining"])) # First 10 ALLOW, then 5 REJECT.
