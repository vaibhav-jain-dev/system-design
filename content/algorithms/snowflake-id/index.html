1 # 1023 MAX_SEQUENCE = (1 << 12) - 1 # 4095 MACHINE_SHIFT = 12 TIMESTAMP_SHIFT = 22 # 10 + 12 class SnowflakeGenerator: """Thread-safe Snowflake ID generator.""" def __init__(self, machine_id: int): if not 0 <= machine_id <= MAX_MACHINE_ID: raise ValueError("machine_id must be 0-1023") self.machine_id = machine_id self.sequence = 0 self.last_timestamp = -1 self.lock = threading.Lock() def _current_millis(self) -> int: return int(time.time() * 1000) def _wait_next_millis(self, last_ts: int) -> int: ts = self._current_millis() while ts <= last_ts: ts = self._current_millis() return ts def generate(self) -> int: with self.lock: ts = self._current_millis() # Guard against clock moving backward if ts < self.last_timestamp: delta = self.last_timestamp - ts if delta < 5: ts = self._wait_next_millis(self.last_timestamp) else: raise Exception("Clock moved backward by {}ms".format(delta)) if ts == self.last_timestamp: self.sequence = (self.sequence + 1) & MAX_SEQUENCE if self.sequence == 0: # Exhausted 4096 IDs this ms ts = self._wait_next_millis(self.last_timestamp) else: self.sequence = 0 # New millisecond, reset self.last_timestamp = ts return (((ts -
